# Phân vùng 1 ổ cứng sdb
parted /dev/sdb mklabel gpt
# Định dạng ổ cứng sdb
parted /dev/sdb mkpart primary ext4 0% 100%
# Show Volum Groups
vgs
# Join /dev/sdb1 vào Volume Groups
vgextend ubuntu-vg /dev/sdb1
# Tăng 100GB cho VG/LV
lvextend -L +100G ubuntu-vg/ubuntu-lv
# Chốt tăng cho LVM OS Ubuntu 
resize2fs /dev/mapper/ubuntu--vg-ubuntu--lv
